\begin{theorem}
For saturated $S$, if $f(a)$ is a unit in $S^{-1}A$, then $a \in S$.
\end{theorem}


\vspace*{10px} 

\noindent
\textit{Proof.}
\[
  \frac{a}{1} \cdot \frac{b}{t} = \frac{1}{1} 
\]

\[
  \frac{ab}{t} = \frac{1}{1}
\]

\[
   (ab, t) \equiv (1, 1)
\]
 
\[
   (ab - t)u = 0
\]

\[
   abu = tu
\]

\[
   abu \in S
\]

As \( S \) is saturated, \( a \in S \).  \qed
